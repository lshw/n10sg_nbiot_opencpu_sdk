void opencpu_pwm_init ( unsigned int  channel,
unsigned int  freq,
unsigned int  ratio 
)

初始化PWM

Parameters
[in]channel通道号，0或3
[in]freq频率
[in]ratio占空比
Returns
Return 空

示例：

//设置占空比，0到100
#define DUTY_RATIO 50
//设置频率
#define PWM_FREQ 2000
//初始化PWM
opencpu_pwm_init(3,PWM_FREQ,DUTY_RATIO);
//开始PWM输出
opencpu_pwm_start(3);
vTaskDelay(1000);
//停止PWM输出
opencpu_pwm_stop(3);
//释放PWM
opencpu_pwm_deinit(3);
